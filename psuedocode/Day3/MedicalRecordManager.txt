INTERFACE Searchable
    METHOD search(criteria)

INTERFACE Printable
    METHOD print()

CLASS LabReport IMPLEMENTS Searchable, Printable
    ATTRIBUTES: patientName, testType, result
    METHOD search(criteria)
        IF testType OR patientName matches criteria
            RETURN report
    METHOD print()
        DISPLAY lab report details

CLASS Prescription IMPLEMENTS Searchable, Printable
    ATTRIBUTES: patientName, medicines, dosage
    METHOD search(criteria)
        IF patientName OR medicine matches criteria
            RETURN prescription
    METHOD print()
        DISPLAY prescription details

CLASS MedicalRecordManager
    ATTRIBUTES: reportsList (list of Searchable and Printable objects)
    METHOD addRecord(record)
        ADD record to reportsList
    METHOD searchAll(criteria)
        FOR each record in reportsList
            CALL record.search(criteria)
    METHOD printAll()
        FOR each record in reportsList
            CALL record.print()

MAIN
    CREATE labReport = new LabReport(patientName, testType, result)
    CREATE prescription = new Prescription(patientName, medicines, dosage)

    CREATE manager = new MedicalRecordManager()
    CALL manager.addRecord(labReport)
    CALL manager.addRecord(prescription)

    CALL manager.searchAll("Blood Test")
    CALL manager.printAll()
END
