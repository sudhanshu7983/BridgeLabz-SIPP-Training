CLASS QuizTimer
    ATTRIBUTES:
        timeSlots   // circular array of integers (seconds per question)
        totalQuestions
        currentIndex

    CONSTRUCTOR(timeSlots)
        this.timeSlots = timeSlots
        this.totalQuestions = LENGTH(timeSlots)
        this.currentIndex = 0

    METHOD startQuiz()
        LOOP i FROM 0 TO totalQuestions-1
            questionTime = timeSlots[currentIndex]
            PRINT "Question " + (i+1) + " time: " + questionTime + " seconds"

            LOOP t FROM questionTime DOWNTO 1
                IF t == 5
                    PRINT "âš  Warning: Only 5 seconds left!"
                WAIT 1 second
            END LOOP

            currentIndex = (currentIndex + 1) MOD totalQuestions

CLASS QuizTimerMain
    METHOD main()
        slots = [30, 45, 60]   // time per question
        timer = QuizTimer(slots)
        timer.startQuiz()
